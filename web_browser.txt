import webbrowser

from speech import speak

def Webbrowser(data):
    data = data.split(" ")
    ask = []
    for counter in range(1, len(data)):
        ask.append(data[counter])
    request = ' '.join(ask)
    speak("Wait a minute. I'm searching for a " + request)
    webbrowser.open('https://google.com?#q=' + request, new=2)
